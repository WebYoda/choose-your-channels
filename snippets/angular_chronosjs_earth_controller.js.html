app.controller("EarthCtrl", ["$scope", "chronosjsChannel", function ($scope, chronosjsChannel) {
    function publishAngle() {
        var data = getAngle(sunImage);
        chronosjsChannel.trigger({
            appName: "earth",
            eventName: "rotation",
            data: data
        });
    }
    setInterval(publishAngle, 200);
}]);
